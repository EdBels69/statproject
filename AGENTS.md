def calculate_cohens_d(group1: pd.Series, group2: pd.Series) -> float:
    """
    Calculate Cohen's d effect size for two groups.
    Returns value between 0 and infinity (0.2=small, 0.5=medium, 0.8=large).
    """
    pooled_std = np.sqrt((group1.var() + group2.var()) / 2)
    return abs(group1.mean() - group2.mean()) / pooled_std
